import MetaTrader5 as mt5
from datetime import datetime, timedelta
import json
import os

class PerformanceTracker:
    """Track and analyze trading performance"""
    
    def __init__(self, stats_file='performance_stats.json'):
        self.stats_file = stats_file
        self.stats = self.load_stats()
    
    def load_stats(self):
        """Load existing stats"""
        if os.path.exists(self.stats_file):
            with open(self.stats_file, 'r') as f:
                return json.load(f)
        return {
            'total_trades': 0,
            'winning_trades': 0,
            'losing_trades': 0,
            'total_profit': 0.0,
            'total_loss': 0.0,
            'best_trade': 0.0,
            'worst_trade': 0.0,
            'trades': []
        }
    
    def save_stats(self):
        """Save stats to file"""
        with open(self.stats_file, 'w') as f:
            json.dump(self.stats, f, indent=2)
    
    def update_from_mt5(self):
        """Update stats from MT5 history"""
        if not mt5.initialize():
            print("❌ MT5 initialization failed")
            return
        
        # Get deals from last 30 days
        from_date = datetime.now() - timedelta(days=30)
        deals = mt5.history_deals_get(from_date, datetime.now())
        
        if not deals:
            print("ℹ️ No deals found")
            mt5.shutdown()
            return
        
        # Filter only closed positions (not deposits/withdrawals)
        trade_deals = [d for d in deals if d.entry == mt5.DEAL_ENTRY_OUT]
        
        for deal in trade_deals:
            # Check if already recorded
            if any(t['ticket'] == deal.ticket for t in self.stats['trades']):
                continue
            
            # Record trade
            trade_data = {
                'ticket': deal.ticket,
                'time': deal.time,
                'symbol': deal.symbol,
                'type': 'BUY' if deal.type == mt5.DEAL_TYPE_BUY else 'SELL',
                'volume': deal.volume,
                'profit': deal.profit,
                'commission': deal.commission,
                'swap': deal.swap
            }
            
            self.stats['trades'].append(trade_data)
            self.stats['total_trades'] += 1
            
            if deal.profit > 0:
                self.stats['winning_trades'] += 1
                self.stats['total_profit'] += deal.profit
                if deal.profit > self.stats['best_trade']:
                    self.stats['best_trade'] = deal.profit
            else:
                self.stats['losing_trades'] += 1
                self.stats['total_loss'] += abs(deal.profit)
                if deal.profit < self.stats['worst_trade']:
                    self.stats['worst_trade'] = deal.profit
        
        self.save_stats()
        mt5.shutdown()
        print(f"✅ Updated stats: {len(trade_deals)} new trades")
    
    def calculate_metrics(self):
        """Calculate performance metrics"""

    }

        return False
    
    print(f"✅ Applying profile: {profile['name']}")
    print(f"   Description: {profile['description']}")
    
    # Update config.py
    import config
    
    config.RISK.CONFLUENCE_THRESHOLD = profile['confluence_threshold']
    config.RISK.AI_CONFIDENCE_THRESHOLD = profile['ai_confidence_threshold']

    config.RISK.MAX_DAILY_TRADES = profile['max_daily_trades']
    
    # Update weights
    for key, value in profile['confluence_weights'].items():
        if hasattr(config.CONFLUENCE_WEIGHTS, key):
            setattr(config.CONFLUENCE_WEIGHTS, key, value)
    
    print("✅ Profile applied successfully")
    return True
