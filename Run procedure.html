import subprocess
import time
import sys
from datetime import datetime

def run_with_restart():
    """Run program with automatic restart on crash"""
    restart_count = 0
    max_restarts = 10  # จำนวนครั้งสูงสุดที่รีสตาร์ทต่อเนื่อง
    restart_delay = 30  # รอ 30 วินาทีก่อนรีสตาร์ท
    
    print("🔄 Auto-Restart Mode Enabled")
    print(f"   Max consecutive restarts: {max_restarts}")
    print(f"   Restart delay: {restart_delay} seconds\n")
    
    while restart_count < max_restarts:
        try:
            print(f"{'='*60}")
            print(f"🚀 Starting HIM System (Attempt {restart_count + 1})")
            print(f"   Time: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
            print(f"{'='*60}\n")
            
            # รันโปรแกรมหลัก
            result = subprocess.run([sys.executable, 'run_continuous.py'])
            
            # ถ้าออกปกติ (exit code 0) = ผู้ใช้หยุดเอง
            if result.returncode == 0:
                print("\n✅ Program exited normally")
                break
            
            # ถ้า crash
            restart_count += 1
            print(f"\n⚠️ Program crashed (Exit code: {result.returncode})")
            print(f"   Restart count: {restart_count}/{max_restarts}")
            
            if restart_count < max_restarts:
                print(f"   Restarting in {restart_delay} seconds...")
                time.sleep(restart_delay)
            else:
                print(f"\n❌ Maximum restart limit reached ({max_restarts})")
                print("   Please check logs for errors")
                break
                
        except KeyboardInterrupt:
            print("\n\n⚠️ Stopped by user (Ctrl+C)")
            break
        except Exception as e:
            print(f"\n❌ Unexpected error: {str(e)}")
            restart_count += 1
            
            if restart_count < max_restarts:
                print(f"   Restarting in {restart_delay} seconds...")
                time.sleep(restart_delay)
            else:
                break
    
    print("\n" + "="*60)
    print("👋 Auto-Restart Mode Terminated")
    print("="*60)

if __name__ == "__main__":
